[docs]def bbox2cbox(bboxes):
    # bboxes: [*, 4]
    sizes = bboxes[..., 2:] - bboxes[..., :2]
    centers = (bboxes[..., :2] + bboxes[..., 2:]) / 2
    cboxes = torch.cat([centers, sizes], dim=-1)
    return cboxes




[docs]def cbox2bbox(cboxes):
    halfs = cboxes[..., 2:] / 2
    lt = cboxes[..., :2] - halfs
    rb = cboxes[..., :2] + halfs
    bboxes = torch.cat([lt, rb], dim=-1)
    return bboxes




[docs]def box_iou_itemwise(bboxes1, bboxes2):
    # bboxes1: [*, 4]
    # bboxes2: [*, 4]
    area1 = box_area(bboxes1)
    area2 = box_area(bboxes2)

    lt = torch.max(bboxes1[..., :2], bboxes2[..., :2])
    rb = torch.min(bboxes1[..., 2:], bboxes2[..., 2:])

    sizes = rb - lt
    sizes = torch.clamp_min(sizes, 0)

    inter_area = torch.prod(sizes, dim=-1)

    return inter_area / (area1 + area2 - inter_area)




[docs]def box_diou_pairwise(bboxes1, bboxes2):
    # bboxes1 [m, 4]
    # bboxes2 [n, 4]

    # Distance-IoU = IoU - pho^2 / c^2
    # where pho is the distance of boxes' center
    # and c is diagonal length of enclosing box

    # calc enclosing box
    lt = torch.min(bboxes1[:, None, :2], bboxes2[None, :, :2])  # [m, n, 2]
    rb = torch.max(bboxes1[:, None, 2:], bboxes2[None, :, 2:])  # [m, n, 2]

    c2 = torch.square(lt - rb).sum(-1)

    ctr1 = (bboxes1[:, 2:] - bboxes1[:, :2]) / 2   # [m, 2]
    ctr2 = (bboxes2[:, 2:] - bboxes2[:, :2]) / 2   # [n, 2]

    pho2 = torch.square(ctr1[:, None, :] - ctr2[None, :, :]).sum(-1)
    diou = box_iou_pairwise - pho2 / c2
    return diou

    # bboxes1: [*, 4]
    # bboxes2: [*, 4]
    lt = torch.min(bboxes1[..., :2], bboxes2[..., :2])
    rb = torch.max(bboxes1[..., 2:], bboxes2[..., 2:])
    c2 = torch.square(lt - rb).sum(-1)
    ctr1 = (bboxes1[:, 2:] - bboxes1[:, :2]) / 2   # 
    ctr2 = (bboxes2[:, 2:] - bboxes2[:, :2]) / 2   # 
    pho2 = torch.square(ctr1 - ctr2).sum(-1)
    diou = box_iou_itemwise(bboxes1, bboxes2) - pho2 / c2
    return diou
